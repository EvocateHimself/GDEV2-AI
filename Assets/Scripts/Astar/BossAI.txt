tree("Root")
    fallback
        tree("FocusPlayer")
        tree("Attack")  
        tree("Defend")       

tree("FocusPlayer")
	while
        sequence 
            playerInRange
            not attackPlayer
		repeat
			sequence
				MoveToPlayer

tree("Attack")
	while
        sequence 
            playerInRange
            attackPlayer
        repeat
            sequence
                ShootFireball
                MoveToPlayer

tree("Defend")
	while defendAllies
        repeat
            fallback
                ActivateShield
                Wait(3.0)
                DeactivateShield
